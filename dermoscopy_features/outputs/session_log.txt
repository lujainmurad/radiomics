   306	def _glcm(patch,levels=32):
   307	    q=np.clip(patch.astype(np.float32)/255.*(levels-1),0,levels-1).astype(np.int32)
   308	    ii,jj=np.mgrid[0:levels,0:levels]
   309	    res={k:[] for k in("contrast","dissimilarity","homogeneity","energy","correlation")}
   310	    for dy,dx in[(0,1),(-1,1),(1,0),(1,1)]:
   311	        r2=slice(max(0,-dy),q.shape[0]-dy if dy>0 else None); r1=slice(dy if dy>0 else max(0,-dy),None if dy<=0 else None)
   312	        c2=slice(max(0,-dx),q.shape[1]-dx if dx>0 else None); c1=slice(dx if dx>0 else max(0,-dx),None if dx<=0 else None)
   313	        a=q[r2,c2].ravel(); b=q[r1,c1].ravel()
   314	        g=np.zeros((levels,levels),np.float64); np.add.at(g,(a,b),1); np.add.at(g,(b,a),1); g/=(g.sum()+1e-9)
   315	        diff=ii-jj
   316	        res["contrast"].append(float((diff**2*g).sum())); res["dissimilarity"].append(float((np.abs(diff)*g).sum()))
   317	        res["homogeneity"].append(float((g/(1+diff**2)).sum())); res["energy"].append(float((g**2).sum()))
   318	        mi_=(ii*g).sum(); mj_=(jj*g).sum()
   319	        si=math.sqrt(float(((ii-mi_)**2*g).sum())+1e-9); sj=math.sqrt(float(((jj-mj_)**2*g).sum())+1e-9)
   320	        res["correlation"].append(float(((ii-mi_)*(jj-mj_)*g).sum()/(si*sj)))
   321	    out={}
   322	    for k,v in res.items(): out[f"glcm_{k}_mean"]=_safe(float(np.mean(v))); out[f"glcm_{k}_std"]=_safe(float(np.std(v)))
   323	    return out
       
   324	def _lbp(gray,bw,n_points=8,radius=1):
   325	    g=gray.astype(np.int32); lbp=np.zeros_like(g)
   326	    for p in range(n_points):
   327	        theta=2*math.pi*p/n_points; dy=int(round(-radius*math.sin(theta))); dx=int(round(radius*math.cos(theta)))
   328	        nb=np.roll(np.roll(g,-dy,axis=0),-dx,axis=1); lbp+=((nb>=g).astype(np.int32)<<p)
   329	    ins=lbp[bw>127].astype(np.float64)
   330	    if ins.size==0: return {"lbp_mean":0.,"lbp_std":0.,"lbp_entropy":0.}
   331	    p=np.histogram(ins,bins=n_points+2,range=(0,255))[0]; p=p/(p.sum()+1e-9)
   332	    return {"lbp_mean":_safe(float(ins.mean())),"lbp_std":_safe(float(ins.std())),
   333	            "lbp_entropy":_safe(-float((p*np.log2(p+1e-9)).sum()))}

   351	    feats=_glcm(patch); feats.update(_lbp(roi_gray,bw)); feats.update(_gabor(roi_gray.astype(np.float32)/255.,bw))
   352	    return feats
       
   353	def evolution_features(bw,area,contours,img_shape):
   354	    img_area=float(img_shape[0]*img_shape[1]); feats={"lesion_to_image_ratio":_safe(area/(img_area+1e-9))}
   355	    if not contours or area==0:
   356	        for k in("bbox_fill_ratio","extent","major_axis","minor_axis","eccentricity"): feats[k]=0.
   357	        return feats
   358	    cnt=max(contours,key=cv2.contourArea); x,y,w,h=cv2.boundingRect(cnt)
   359	    feats["bbox_fill_ratio"]=_safe(area/img_area); feats["extent"]=_safe(area/(float(w*h)+1e-9))
   360	    if len(cnt)>=5:
   361	        (_,_),(ma,mi),_=cv2.fitEllipse(cnt)
   362	        feats["major_axis"]=_safe(float(ma)); feats["minor_axis"]=_safe(float(mi))
   363	        feats["eccentricity"]=_safe(math.sqrt(max(0.,1-(mi/(ma+1e-9))**2)))
   364	    else: feats["major_axis"]=feats["minor_axis"]=feats["eccentricity"]=0.
   365	    return feats
       
   366	def histogram_radiomics(roi_gray,bw):
   367	    vals=roi_gray[bw.astype(bool)].astype(np.float64)
   368	    z={k:0. for k in("hist_energy","hist_entropy","hist_mean","hist_variance","hist_kurtosis",
   369	        "hist_skewness","hist_p10","hist_p25","hist_p75","hist_p90","hist_iqr","hist_range",
   370	        "hist_rms","hist_uniformity","hist_median","hist_mad")}
   371	    if vals.size==0: return z
   372	    p=np.histogram(vals,bins=128,range=(0,255))[0]; p=p/(p.sum()+1e-9)
   373	    mu=vals.mean(); sig=vals.std()+1e-9; p10,p25,p75,p90=np.percentile(vals,[10,25,75,90])
   374	    return {"hist_energy":_safe(float((p**2).sum())),"hist_entropy":_safe(float(-(p*np.log2(p+1e-9)).sum())),
   375	            "hist_mean":_safe(float(mu)),"hist_variance":_safe(float(vals.var())),
   376	            "hist_kurtosis":_safe(float(((vals-mu)**4).mean()/sig**4)),
   377	            "hist_skewness":_safe(float(((vals-mu)**3).mean()/sig**3)),
   378	            "hist_p10":_safe(float(p10)),"hist_p25":_safe(float(p25)),
   379	            "hist_p75":_safe(float(p75)),"hist_p90":_safe(float(p90)),
   380	            "hist_iqr":_safe(float(p75-p25)),"hist_range":_safe(float(vals.max()-vals.min())),
   381	            "hist_rms":_safe(float(math.sqrt((vals**2).mean()))),"hist_uniformity":_safe(float((p**2).sum())),
   382	            "hist_median":_safe(float(np.median(vals))),"hist_mad":_safe(float(np.abs(vals-mu).mean()))}
       
   383	def shape_radiomics(bw,area,contours):
   384	    z={k:0. for k in("shape_perimeter","shape_area","shape_sphericity","shape_roundness",
   385	        "shape_pd_ratio","shape_max_diameter","shape_min_diameter")}
   386	    if not contours or area==0: return z
   387	    cnt=max(contours,key=cv2.contourArea); peri=cv2.arcLength(cnt,True)+1e-9
   388	    feats={"shape_perimeter":_safe(peri),"shape_area":_safe(area),
   389	           "shape_sphericity":_safe(4*math.pi*area/peri**2),
   390	           "shape_roundness":_safe(4*area/(math.pi*(peri/math.pi)**2)),
   391	           "shape_pd_ratio":_safe(peri/(2*math.sqrt(math.pi*area)))}
   392	    pts=cv2.convexHull(cnt)[:,0,:]
   393	    if len(pts)>=2:
   394	        d=np.sqrt(((pts[:,None]-pts[None,:])**2).sum(-1))
   395	        feats["shape_max_diameter"]=_safe(float(d.max()))
   396	        pos=d[d>0]; feats["shape_min_diameter"]=_safe(float(pos.min()) if pos.size else 0.)
   397	    else: feats["shape_max_diameter"]=feats["shape_min_diameter"]=0.
   398	    return feats
       
   399	def hu_moments_features(bw):
   400	    hu=cv2.HuMoments(cv2.moments(bw)).flatten()

   402	def fourier_descriptor_features(contours,n_desc=10):
   403	    feats={f"fourier_desc_{i+1}":0. for i in range(n_desc)}
   404	    if not contours: return feats
   405	    cnt=max(contours,key=cv2.contourArea)[:,0,:]
   406	    if len(cnt)<n_desc*2: return feats
   407	    z=cnt[:,0].astype(np.float64)+1j*cnt[:,1].astype(np.float64)
   408	    fd=np.fft.fft(z); norm=abs(fd[1])+1e-9
   409	    for i,m in enumerate(np.abs(fd[1:n_desc+1])/norm): feats[f"fourier_desc_{i+1}"]=_safe(float(m))
   410	    return feats
       
   411	def glrlm_features(roi_gray,bw,levels=32,max_run=64):
   412	    area=bw.sum()
   413	    z={k:0. for k in("glrlm_sre","glrlm_lre","glrlm_gln","glrlm_rln","glrlm_rp","glrlm_lgre","glrlm_hgre")}
   414	    if area==0: return z
   415	    q=np.clip(roi_gray.astype(np.float32)/255.*(levels-1),0,levels-1).astype(np.int32)
   416	    glrlm=np.zeros((levels,max_run),np.float64); H,W=q.shape
   417	    for row in range(H):
   418	        rq=q[row]; rb=bw[row]
   419	        if not rb.any(): continue
   420	        vals=rq*rb; changes=np.diff(vals,prepend=-(levels+1))!=0; mon=np.diff(rb.astype(np.int8),prepend=0)>0
   421	        starts=np.where((changes|mon)&(rb>0))[0]; ends=np.append(starts[1:],W)
   422	        for s,e in zip(starts,ends):
   423	            sl=rb[s:e]; off=np.argmin(sl)
   424	            ln=off if (not sl[off]) and off>0 else len(sl)
   425	            glrlm[rq[s],min(ln,max_run)-1]+=1
   426	    total=glrlm.sum()+1e-9
   427	    RL=np.arange(1,max_run+1,dtype=np.float64)[np.newaxis,:]; GL=np.arange(1,levels+1,dtype=np.float64)[:,np.newaxis]
   428	    return {"glrlm_sre":_safe(float((glrlm/(RL**2)).sum()/total)),"glrlm_lre":_safe(float((glrlm*RL**2).sum()/total)),
   429	            "glrlm_gln":_safe(float((glrlm.sum(1)**2).sum()/total)),"glrlm_rln":_safe(float((glrlm.sum(0)**2).sum()/total)),
   430	            "glrlm_rp":_safe(float(total/area)),"glrlm_lgre":_safe(float((glrlm/(GL**2)).sum()/total)),
   431	            "glrlm_hgre":_safe(float((glrlm*GL**2).sum()/total))}
       
   432	def extract_all(masked_img_path, mask_path, max_dim=DEFAULT_MAX_DIM):
   433	    img_bgr=cv2.imread(masked_img_path)
   434	    if img_bgr is None: raise FileNotFoundError(f"Cannot read: {masked_img_path}")
   435	    mask_raw=cv2.imread(mask_path,0)
   436	    if mask_raw is None: raise FileNotFoundError(f"Cannot read: {mask_path}")
   437	    roi_rgb=cv2.cvtColor(img_bgr,cv2.COLOR_BGR2RGB)
   438	    mask_bin=cv2.resize(mask_raw,(roi_rgb.shape[1],roi_rgb.shape[0]),interpolation=cv2.INTER_NEAREST)
   439	    roi_rgb_s,mask_s=_resize_pair(roi_rgb,mask_bin,max_dim)
   440	    roi_gray_s=cv2.cvtColor(cv2.cvtColor(roi_rgb_s,cv2.COLOR_RGB2BGR),cv2.COLOR_BGR2GRAY)
   441	    bw=(mask_s>127).astype(np.uint8); area=float(bw.sum())
   442	    contours,_=cv2.findContours(bw,cv2.RETR_EXTERNAL,cv2.CHAIN_APPROX_SIMPLE)
   443	    record={"image_id":_stem(masked_img_path)}
   444	    record.update(asymmetry_features(bw,area,contours))
   445	    record.update(border_features(bw,area,contours,roi_gray_s))
   446	    record.update(color_features(roi_rgb_s,bw))
   447	    record.update(texture_features(roi_gray_s,bw))
   448	    record.update(evolution_features(bw,area,contours,roi_rgb_s.shape[:2]))
   449	    record.update(histogram_radiomics(roi_gray_s,bw))
   450	    record.update(shape_radiomics(bw,area,contours))
   451	    record.update(glrlm_features(roi_gray_s,bw))
   452	    record.update(hu_moments_features(bw))
   453	    record.update(fourier_descriptor_features(contours))
   454	    record.update(pyradiomics_features(roi_gray_s,bw))
   455	    return record

   458	    split,img_path,mask_path=args

   460	        rec=extract_all(img_path,mask_path,max_dim=_WORKER_MAX_DIM)

   463	def discover_pairs(masked_root,mask_root):
   464	    entries=os.listdir(masked_root)
   465	    subdirs=[e for e in entries if os.path.isdir(os.path.join(masked_root,e))]
   466	    splits=subdirs if subdirs else [""]
   467	    pairs=[]
   468	    for split in splits:
   469	        img_dir=os.path.join(masked_root,split) if split else masked_root
   470	        msk_dir=os.path.join(mask_root,split) if split else mask_root
   471	        if not os.path.isdir(msk_dir):
   472	            print(f"  [WARN] mask dir not found: {msk_dir}"); continue
   473	        img_files=sorted(p for ext in("*.png","*.jpg","*.jpeg") for p in glob(os.path.join(img_dir,ext)))
   474	        mask_map={_stem(mp):mp for mp in glob(os.path.join(msk_dir,"*.png"))+glob(os.path.join(msk_dir,"*.jpg"))}
   475	        for ip in img_files:
   476	            stem=_stem(ip); base=stem.replace("_masked","")
   477	            key=base if base in mask_map else (stem if stem in mask_map else None)
   478	            if key: pairs.append((split or "unknown",ip,mask_map[key]))
   479	    print(f"  Found {len(pairs)} pairs across splits: {sorted(set(p[0] for p in pairs))}")
   480	    return pairs
       
   481	def main():
   482	    parser=argparse.ArgumentParser()
   483	    parser.add_argument("--masked_root",default=DEFAULT_MASKED_ROOT)
   484	    parser.add_argument("--mask_root",default=DEFAULT_MASK_ROOT)
   485	    parser.add_argument("--output",default=DEFAULT_OUTPUT_CSV)
   486	    parser.add_argument("--max_dim",type=int,default=DEFAULT_MAX_DIM)
   487	    parser.add_argument("--workers",type=int,default=DEFAULT_WORKERS)
   488	    args=parser.parse_args()
   489	    global _WORKER_MAX_DIM; _WORKER_MAX_DIM=args.max_dim
   490	    print("="*62)
   491	    print(f"  masked_root : {args.masked_root}")
   492	    print(f"  mask_root   : {args.mask_root}")
   493	    print(f"  output      : {args.output}")
   494	    print(f"  max_dim={args.max_dim}  workers={args.workers}  PyRadiomics={'ON' if _RADIOMICS_OK else 'OFF'}")
   495	    print("="*62)
   496	    pairs=discover_pairs(args.masked_root,args.mask_root)
   497	    if not pairs: raise RuntimeError("No paired images found.")
   498	    rows,errors=[],[]
   499	    if args.workers>1:
   500	        with Pool(args.workers) as pool:
   501	            for rec,err in tqdm(pool.imap_unordered(_worker,pairs,chunksize=4),total=len(pairs),desc="Extracting"):
   502	                (rows if rec is not None else errors).append(rec or err)
   503	    else:
   504	        for item in tqdm(pairs,desc="Extracting"):
   505	            rec,err=_worker(item)
   506	            (rows if rec is not None else errors).append(rec or err)
   507	    if errors:
   508	        print(f"\n  WARNING: {len(errors)} failed:")
   509	        for e in errors[:10]:
   510	            if isinstance(e,tuple): print(f"    {os.path.basename(e[0])}: {e[1]}")
   511	    if not rows: raise RuntimeError("No features extracted.")
   512	    all_keys=list(rows[0].keys())
   513	    for r in rows:
   514	        for k in all_keys: r.setdefault(k,0.0)
   515	    os.makedirs(os.path.dirname(os.path.abspath(args.output)),exist_ok=True)
   516	    with open(args.output,"w",newline="") as f:
   517	        w=csv.DictWriter(f,fieldnames=all_keys); w.writeheader(); w.writerows(rows)
   518	    print(f"\n  ✓  {len(rows)} rows × {len(all_keys)} columns → {args.output}")

   587	def asymmetry_features(bw, area, contours):
   588	    if area==0: return {k:0.0 for k in("asym_axis_aligned","asym_pca","compactness","elongation","axis_ratio")}
   589	    feats={}; M=cv2.moments(bw); cx=M["m10"]/(M["m00"]+1e-9); cy=M["m01"]/(M["m00"]+1e-9)

   603	def _fractal(bw):
   604	    edge=cv2.Canny((bw*255).astype(np.uint8),50,150); sizes,counts=[],[]
   605	    s=2
   606	    while s<=128:
   607	        h2=(edge.shape[0]//s)*s; w2=(edge.shape[1]//s)*s
   608	        if h2==0 or w2==0: s*=2; continue
   609	        nb=edge[:h2,:w2].reshape(h2//s,s,w2//s,s).any(axis=(1,3)).sum()
   610	        if nb>0: sizes.append(s); counts.append(nb)
   611	        s*=2
   612	    if len(sizes)<2: return 1.0
   613	    return _safe(np.polyfit(np.log(1./np.array(sizes,np.float64)),np.log(np.array(counts,np.float64)),1)[0])

   649	    feats["color_white_frac"]=_safe(float(np.all(px>200,1).sum())/n)
   650	    feats["color_dark_frac"]=_safe(float(np.all(px<50,1).sum())/n)
   651	    lum=0.299*px[:,0].astype(float)+0.587*px[:,1]+0.114*px[:,2]
   652	    feats["color_dominant_spread"]=_safe(float(np.percentile(lum,75)-np.percentile(lum,25)))
   653	    roi_bgr=cv2.cvtColor(roi_rgb,cv2.COLOR_RGB2BGR); bw_u8=(bw*255).astype(np.uint8); nc=0
   654	    for k,(lo,hi) in _ABCDE_COLORS.items():
   655	        cm=cv2.inRange(roi_bgr,np.array(lo,np.uint8),np.array(hi,np.uint8))
   656	        cm=cv2.bitwise_and(cm,cm,mask=bw_u8)
   657	        frac=float(cm.sum()/255)/n; feats[k]=_safe(frac)
   658	        if frac>0.01: nc+=1
   659	    feats["color_n_abcde_colors"]=float(nc)
   660	    return feats
       
   661	def _glcm(patch,levels=32):
   662	    q=np.clip(patch.astype(np.float32)/255.*(levels-1),0,levels-1).astype(np.int32)
   663	    ii,jj=np.mgrid[0:levels,0:levels]
   664	    res={k:[] for k in("contrast","dissimilarity","homogeneity","energy","correlation")}
   665	    for dy,dx in[(0,1),(-1,1),(1,0),(1,1)]:
   666	        r2=slice(max(0,-dy),q.shape[0]-dy if dy>0 else None); r1=slice(dy if dy>0 else max(0,-dy),None if dy<=0 else None)
   667	        c2=slice(max(0,-dx),q.shape[1]-dx if dx>0 else None); c1=slice(dx if dx>0 else max(0,-dx),None if dx<=0 else None)
   668	        a=q[r2,c2].ravel(); b=q[r1,c1].ravel()
   669	        g=np.zeros((levels,levels),np.float64); np.add.at(g,(a,b),1); np.add.at(g,(b,a),1); g/=(g.sum()+1e-9)
   670	        diff=ii-jj
   671	        res["contrast"].append(float((diff**2*g).sum())); res["dissimilarity"].append(float((np.abs(diff)*g).sum()))
   672	        res["homogeneity"].append(float((g/(1+diff**2)).sum())); res["energy"].append(float((g**2).sum()))
   673	        mi_=(ii*g).sum(); mj_=(jj*g).sum()
   674	        si=math.sqrt(float(((ii-mi_)**2*g).sum())+1e-9); sj=math.sqrt(float(((jj-mj_)**2*g).sum())+1e-9)
   675	        res["correlation"].append(float(((ii-mi_)*(jj-mj_)*g).sum()/(si*sj)))
   676	    out={}
   677	    for k,v in res.items(): out[f"glcm_{k}_mean"]=_safe(float(np.mean(v))); out[f"glcm_{k}_std"]=_safe(float(np.std(v)))
   678	    return out
       
   679	def _lbp(gray,bw,n_points=8,radius=1):
   680	    g=gray.astype(np.int32); lbp=np.zeros_like(g)
   681	    for p in range(n_points):
   682	        theta=2*math.pi*p/n_points; dy=int(round(-radius*math.sin(theta))); dx=int(round(radius*math.cos(theta)))
   683	        nb=np.roll(np.roll(g,-dy,axis=0),-dx,axis=1); lbp+=((nb>=g).astype(np.int32)<<p)
   684	    ins=lbp[bw>127].astype(np.float64)
   685	    if ins.size==0: return {"lbp_mean":0.,"lbp_std":0.,"lbp_entropy":0.}
   686	    p=np.histogram(ins,bins=n_points+2,range=(0,255))[0]; p=p/(p.sum()+1e-9)
   687	    return {"lbp_mean":_safe(float(ins.mean())),"lbp_std":_safe(float(ins.std())),
   688	            "lbp_entropy":_safe(-float((p*np.log2(p+1e-9)).sum()))}
       
   689	def _gabor(gray_f32,bw):
   690	    feats={}
   691	    for sc in(3,5,9):
   692	        for ang in(0,45,90,135):
   693	            kern=cv2.getGaborKernel((sc*2+1,sc*2+1),sc*.56,math.radians(ang),sc*1.2,.5,0,cv2.CV_32F)
   694	            ins=cv2.filter2D(gray_f32,cv2.CV_32F,kern)[bw>127]
   695	            feats[f"gabor_s{sc}_a{ang}_energy"]=_safe(float((ins**2).mean())) if ins.size else 0.
   696	    return feats
       
   697	def texture_features(roi_gray,bw):
   698	    area=bw.sum()
   699	    z={f"glcm_{k}_{s}":0. for k in("contrast","dissimilarity","homogeneity","energy","correlation") for s in("mean","std")}
   700	    if area==0:
   701	        z.update(lbp_mean=0.,lbp_std=0.,lbp_entropy=0.)
   702	        for sc in(3,5,9):
   703	            for a in(0,45,90,135): z[f"gabor_s{sc}_a{a}_energy"]=0.
   704	        return z
   705	    ys,xs=np.where(bw>0); patch=roi_gray[ys.min():ys.max()+1,xs.min():xs.max()+1]
   706	    feats=_glcm(patch); feats.update(_lbp(roi_gray,bw)); feats.update(_gabor(roi_gray.astype(np.float32)/255.,bw))
   707	    return feats
       
   708	def evolution_features(bw,area,contours,img_shape):
   709	    img_area=float(img_shape[0]*img_shape[1]); feats={"lesion_to_image_ratio":_safe(area/(img_area+1e-9))}
   710	    if not contours or area==0:
   711	        for k in("bbox_fill_ratio","extent","major_axis","minor_axis","eccentricity"): feats[k]=0.
   712	        return feats
   713	    cnt=max(contours,key=cv2.contourArea); x,y,w,h=cv2.boundingRect(cnt)
   714	    feats["bbox_fill_ratio"]=_safe(area/img_area); feats["extent"]=_safe(area/(float(w*h)+1e-9))
   715	    if len(cnt)>=5:
   716	        (_,_),(ma,mi),_=cv2.fitEllipse(cnt)
   717	        feats["major_axis"]=_safe(float(ma)); feats["minor_axis"]=_safe(float(mi))
   718	        feats["eccentricity"]=_safe(math.sqrt(max(0.,1-(mi/(ma+1e-9))**2)))
   719	    else: feats["major_axis"]=feats["minor_axis"]=feats["eccentricity"]=0.
   720	    return feats
       
   721	def histogram_radiomics(roi_gray,bw):
   722	    vals=roi_gray[bw.astype(bool)].astype(np.float64)
   723	    z={k:0. for k in("hist_energy","hist_entropy","hist_mean","hist_variance","hist_kurtosis",
   724	        "hist_skewness","hist_p10","hist_p25","hist_p75","hist_p90","hist_iqr","hist_range",
   725	        "hist_rms","hist_uniformity","hist_median","hist_mad")}
   726	    if vals.size==0: return z
   727	    p=np.histogram(vals,bins=128,range=(0,255))[0]; p=p/(p.sum()+1e-9)
   728	    mu=vals.mean(); sig=vals.std()+1e-9; p10,p25,p75,p90=np.percentile(vals,[10,25,75,90])
   729	    return {"hist_energy":_safe(float((p**2).sum())),"hist_entropy":_safe(float(-(p*np.log2(p+1e-9)).sum())),
   730	            "hist_mean":_safe(float(mu)),"hist_variance":_safe(float(vals.var())),
   731	            "hist_kurtosis":_safe(float(((vals-mu)**4).mean()/sig**4)),
   732	            "hist_skewness":_safe(float(((vals-mu)**3).mean()/sig**3)),
   733	            "hist_p10":_safe(float(p10)),"hist_p25":_safe(float(p25)),
   734	            "hist_p75":_safe(float(p75)),"hist_p90":_safe(float(p90)),
   735	            "hist_iqr":_safe(float(p75-p25)),"hist_range":_safe(float(vals.max()-vals.min())),
   736	            "hist_rms":_safe(float(math.sqrt((vals**2).mean()))),"hist_uniformity":_safe(float((p**2).sum())),
   737	            "hist_median":_safe(float(np.median(vals))),"hist_mad":_safe(float(np.abs(vals-mu).mean()))}
       
   738	def shape_radiomics(bw,area,contours):
   739	    z={k:0. for k in("shape_perimeter","shape_area","shape_sphericity","shape_roundness",
   740	        "shape_pd_ratio","shape_max_diameter","shape_min_diameter")}
   741	    if not contours or area==0: return z
   742	    cnt=max(contours,key=cv2.contourArea); peri=cv2.arcLength(cnt,True)+1e-9
   743	    feats={"shape_perimeter":_safe(peri),"shape_area":_safe(area),
   744	           "shape_sphericity":_safe(4*math.pi*area/peri**2),
   745	           "shape_roundness":_safe(4*area/(math.pi*(peri/math.pi)**2)),
   746	           "shape_pd_ratio":_safe(peri/(2*math.sqrt(math.pi*area)))}
   747	    pts=cv2.convexHull(cnt)[:,0,:]
   748	    if len(pts)>=2:
   749	        d=np.sqrt(((pts[:,None]-pts[None,:])**2).sum(-1))
   750	        feats["shape_max_diameter"]=_safe(float(d.max()))
   751	        pos=d[d>0]; feats["shape_min_diameter"]=_safe(float(pos.min()) if pos.size else 0.)
   752	    else: feats["shape_max_diameter"]=feats["shape_min_diameter"]=0.
   753	    return feats
       
   754	def hu_moments_features(bw):
   755	    hu=cv2.HuMoments(cv2.moments(bw)).flatten()
   756	    return {f"hu_moment_{i+1}":_safe(-math.copysign(1,h)*math.log10(abs(h)+1e-15)) for i,h in enumerate(hu)}
       
   757	def fourier_descriptor_features(contours,n_desc=10):
   758	    feats={f"fourier_desc_{i+1}":0. for i in range(n_desc)}
   759	    if not contours: return feats
   760	    cnt=max(contours,key=cv2.contourArea)[:,0,:]
   761	    if len(cnt)<n_desc*2: return feats
   762	    z=cnt[:,0].astype(np.float64)+1j*cnt[:,1].astype(np.float64)
   763	    fd=np.fft.fft(z); norm=abs(fd[1])+1e-9
   764	    for i,m in enumerate(np.abs(fd[1:n_desc+1])/norm): feats[f"fourier_desc_{i+1}"]=_safe(float(m))
   765	    return feats
       
   766	def glrlm_features(roi_gray,bw,levels=32,max_run=64):
   767	    area=bw.sum()
   768	    z={k:0. for k in("glrlm_sre","glrlm_lre","glrlm_gln","glrlm_rln","glrlm_rp","glrlm_lgre","glrlm_hgre")}
   769	    if area==0: return z
   770	    q=np.clip(roi_gray.astype(np.float32)/255.*(levels-1),0,levels-1).astype(np.int32)
   771	    glrlm=np.zeros((levels,max_run),np.float64); H,W=q.shape
   772	    for row in range(H):
   773	        rq=q[row]; rb=bw[row]
   774	        if not rb.any(): continue
   775	        vals=rq*rb; changes=np.diff(vals,prepend=-(levels+1))!=0; mon=np.diff(rb.astype(np.int8),prepend=0)>0
   776	        starts=np.where((changes|mon)&(rb>0))[0]; ends=np.append(starts[1:],W)
   777	        for s,e in zip(starts,ends):
   778	            sl=rb[s:e]; off=np.argmin(sl)
   779	            ln=off if (not sl[off]) and off>0 else len(sl)
   780	            glrlm[rq[s],min(ln,max_run)-1]+=1
   781	    total=glrlm.sum()+1e-9
   782	    RL=np.arange(1,max_run+1,dtype=np.float64)[np.newaxis,:]; GL=np.arange(1,levels+1,dtype=np.float64)[:,np.newaxis]
   783	    return {"glrlm_sre":_safe(float((glrlm/(RL**2)).sum()/total)),"glrlm_lre":_safe(float((glrlm*RL**2).sum()/total)),
   784	            "glrlm_gln":_safe(float((glrlm.sum(1)**2).sum()/total)),"glrlm_rln":_safe(float((glrlm.sum(0)**2).sum()/total)),
   785	            "glrlm_rp":_safe(float(total/area)),"glrlm_lgre":_safe(float((glrlm/(GL**2)).sum()/total)),
   786	            "glrlm_hgre":_safe(float((glrlm*GL**2).sum()/total))}
       
   787	def extract_all(masked_img_path, mask_path, max_dim=DEFAULT_MAX_DIM):
   788	    img_bgr=cv2.imread(masked_img_path)
   789	    if img_bgr is None: raise FileNotFoundError(f"Cannot read: {masked_img_path}")
   790	    mask_raw=cv2.imread(mask_path,0)
   791	    if mask_raw is None: raise FileNotFoundError(f"Cannot read: {mask_path}")
   792	    roi_rgb=cv2.cvtColor(img_bgr,cv2.COLOR_BGR2RGB)
   793	    mask_bin=cv2.resize(mask_raw,(roi_rgb.shape[1],roi_rgb.shape[0]),interpolation=cv2.INTER_NEAREST)
   794	    roi_rgb_s,mask_s=_resize_pair(roi_rgb,mask_bin,max_dim)
   795	    roi_gray_s=cv2.cvtColor(cv2.cvtColor(roi_rgb_s,cv2.COLOR_RGB2BGR),cv2.COLOR_BGR2GRAY)
   796	    bw=(mask_s>127).astype(np.uint8); area=float(bw.sum())
   797	    contours,_=cv2.findContours(bw,cv2.RETR_EXTERNAL,cv2.CHAIN_APPROX_SIMPLE)
   798	    record={"image_id":_stem(masked_img_path)}
   799	    record.update(asymmetry_features(bw,area,contours))
   800	    record.update(border_features(bw,area,contours,roi_gray_s))
   801	    record.update(color_features(roi_rgb_s,bw))
   802	    record.update(texture_features(roi_gray_s,bw))
   803	    record.update(evolution_features(bw,area,contours,roi_rgb_s.shape[:2]))
   804	    record.update(histogram_radiomics(roi_gray_s,bw))
   805	    record.update(shape_radiomics(bw,area,contours))
   806	    record.update(glrlm_features(roi_gray_s,bw))
   807	    record.update(hu_moments_features(bw))
   808	    record.update(fourier_descriptor_features(contours))
   809	    record.update(pyradiomics_features(roi_gray_s,bw))
   810	    return record
       
   811	_WORKER_MAX_DIM = DEFAULT_MAX_DIM
   812	def _worker(args):
   813	    split,img_path,mask_path=args
   814	    try:
   815	        rec=extract_all(img_path,mask_path,max_dim=_WORKER_MAX_DIM)

   818	def discover_pairs(masked_root,mask_root):
   819	    entries=os.listdir(masked_root)
   820	    subdirs=[e for e in entries if os.path.isdir(os.path.join(masked_root,e))]
   821	    splits=subdirs if subdirs else [""]
   822	    pairs=[]
   823	    for split in splits:
   824	        img_dir=os.path.join(masked_root,split) if split else masked_root
   825	        msk_dir=os.path.join(mask_root,split) if split else mask_root
   826	        if not os.path.isdir(msk_dir):
   827	            print(f"  [WARN] mask dir not found: {msk_dir}"); continue
   828	        img_files=sorted(p for ext in("*.png","*.jpg","*.jpeg") for p in glob(os.path.join(img_dir,ext)))
   829	        mask_map={_stem(mp):mp for mp in glob(os.path.join(msk_dir,"*.png"))+glob(os.path.join(msk_dir,"*.jpg"))}
   830	        for ip in img_files:
   831	            stem=_stem(ip); base=stem.replace("_masked","")
   832	            key=base if base in mask_map else (stem if stem in mask_map else None)
   833	            if key: pairs.append((split or "unknown",ip,mask_map[key]))
   834	    print(f"  Found {len(pairs)} pairs across splits: {sorted(set(p[0] for p in pairs))}")
   835	    return pairs
       
   836	def main():
   837	    parser=argparse.ArgumentParser()
   838	    parser.add_argument("--masked_root",default=DEFAULT_MASKED_ROOT)
   839	    parser.add_argument("--mask_root",default=DEFAULT_MASK_ROOT)
   840	    parser.add_argument("--output",default=DEFAULT_OUTPUT_CSV)
   841	    parser.add_argument("--max_dim",type=int,default=DEFAULT_MAX_DIM)
   842	    parser.add_argument("--workers",type=int,default=DEFAULT_WORKERS)
   843	    args=parser.parse_args()
   844	    global _WORKER_MAX_DIM; _WORKER_MAX_DIM=args.max_dim
   845	    print("="*62)
   846	    print(f"  masked_root : {args.masked_root}")
   847	    print(f"  mask_root   : {args.mask_root}")
   848	    print(f"  output      : {args.output}")
   849	    print(f"  max_dim={args.max_dim}  workers={args.workers}  PyRadiomics={'ON' if _RADIOMICS_OK else 'OFF'}")
   850	    print("="*62)
   851	    pairs=discover_pairs(args.masked_root,args.mask_root)
   852	    if not pairs: raise RuntimeError("No paired images found.")
   853	    rows,errors=[],[]
   854	    if args.workers>1:
   855	        with Pool(args.workers) as pool:
   856	            for rec,err in tqdm(pool.imap_unordered(_worker,pairs,chunksize=4),total=len(pairs),desc="Extracting"):
   857	                (rows if rec is not None else errors).append(rec or err)
   858	    else:
   859	        for item in tqdm(pairs,desc="Extracting"):
   860	            rec,err=_worker(item)
   861	            (rows if rec is not None else errors).append(rec or err)
   862	    if errors:
   863	        print(f"\n  WARNING: {len(errors)} failed:")
   864	        for e in errors[:10]:
   865	            if isinstance(e,tuple): print(f"    {os.path.basename(e[0])}: {e[1]}")
   866	    if not rows: raise RuntimeError("No features extracted.")
   867	    all_keys=list(rows[0].keys())
   868	    for r in rows:
   869	        for k in all_keys: r.setdefault(k,0.0)
   870	    os.makedirs(os.path.dirname(os.path.abspath(args.output)),exist_ok=True)
   871	    with open(args.output,"w",newline="") as f:
   872	        w=csv.DictWriter(f,fieldnames=all_keys); w.writeheader(); w.writerows(rows)
   873	    print(f"\n  ✓  {len(rows)} rows × {len(all_keys)} columns → {args.output}")

   875	    main()
